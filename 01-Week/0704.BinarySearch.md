### 我的答案
```Go
func search(nums []int, target int) int {
    start, end := 0, len(nums)-1
    for i := 0; i < len(nums); i++ {
        mid := (end-start)/2 + start
        if nums[mid] == target {
            return mid
        } else if nums[mid] > target {
            end = mid
        } else {
            start = mid
        }
    }
    return -1
}
```

### 官方題解
#### 本題可以使用『二分查找』解題
```Go
func search(nums []int, target int) int {
    left, right := 0, len(nums)-1
    for left <= right {
        mid := (right-left)/2 + left
        num := nums[mid]
        if num == target {
            return mid
        } else if num > target {
            right = mid - 1
        } else {
            left = mid + 1
        }
    }
    return -1
}
```
